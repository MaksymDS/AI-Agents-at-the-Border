% amber tick + short blue rule (openers)

% ---------------------------------------------------------------- headings
\setkomafont{section}{\sffamily\bfseries\fontsize{13}{16}\selectfont\color{customsblue}}
\setkomafont{subsection}{\sffamily\bfseries\fontsize{11}{13.5}\selectfont\color{inkbody}}
\RedeclareSectionCommand[beforeskip=18pt plus 4pt, afterskip=6pt]{section}
\RedeclareSectionCommand[beforeskip=12pt plus 3pt, afterskip=3pt]{subsection}
% Keep a heading with enough following text to establish the section. This
% eliminates stranded section titles and Field-note labels at page bottoms.
\AddToHook{cmd/section/before}{\Needspace{6\baselineskip}}
\AddToHook{cmd/subsection/before}{\Needspace{4\baselineskip}}

% ---------------------------------------------------------------- chapter opener
\renewcommand\chapterlinesformat[3]{%
  \Ifstr{#2}{}{% unnumbered (front matter): title + tick only
    {\raggedright\sffamily\bfseries\fontsize{22}{26}\selectfont\color{customsblue}#3\par}%
    \vskip 6pt\tickrule\par
  }{%
    {\fontsize{10}{12}\selectfont\color{slate}\labelfont{\chapapp~\thechapter}\par}%
    \vskip 12pt
    \noindent\begin{minipage}[b]{0.76\textwidth}
      \raggedright\sffamily\bfseries\fontsize{22}{26}\selectfont\color{customsblue}#3
    \end{minipage}%
    \hfill
    \begin{minipage}[b]{0.2\textwidth}
      \raggedleft\interlight\fontsize{84}{84}\selectfont\color{rampltwo}\thechapter
    \end{minipage}\par
    \vskip 8pt\tickrule\par
  }%
}

% ---------------------------------------------------------------- part opener
% Designed progressive-network part openers (COVER-SPEC section 6):
% each \part becomes the corresponding full-page SVG (prebuilt to PDF),
% with a TOC entry and the centred folio of the plain style. The
% cmd/part/before hook (mainmatter switch, endnote flush) still fires.
\usepackage{pdfpages}
\RenewDocumentCommand\part{om}{%
  % the generic cmd/part/before hook does not survive this redefinition:
  % flush pending endnotes and switch to mainmatter explicitly
  \flushchapternotes
  \ifbookmainmatter\else\bookrealmainmatter\bookmainmattertrue\fi
  \clearpage
  \refstepcounter{part}%
  \addcontentsline{toc}{part}{\protect\numberline{\thepart}#2}%
  \includepdf[pagecommand={\thispagestyle{plain}}]{assets/prebuilt/interior-part-\arabic{part}.pdf}%
}
