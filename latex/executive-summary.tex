% Repeated scan layer at the start of every brief chapter. The three-column
% grammar lets a reader identify the decision, the evidence and the stopping
% condition before reading the narrative.
\newcommand{\ExecutiveStripCell}[3]{%
  \begin{minipage}[t]{\linewidth}
  {\fontsize{7.6}{9}\selectfont\color{#1}\labelfontbold{#2}}\par\vspace{4pt}
  {\fontsize{8.7}{11.2}\selectfont\color{inkbody}#3}\par
  \end{minipage}}
\newcommand{\ExecutiveDecisionStrip}[3]{%
  \begin{tcolorbox}[enhanced,boxrule=.7pt,colframe=hairlinegray,
    colback=paneltint,arc=2pt,left=9pt,right=9pt,top=9pt,bottom=9pt,
    before skip=10pt,after skip=16pt]
  \begin{tabularx}{\linewidth}{@{}X!{\color{hairlinegray}\vrule width .7pt}X!{\color{hairlinegray}\vrule width .7pt}X@{}}

    \ExecutiveStripCell{slate}{Evidence}{#2} &
